        <textarea id="postContent" rows="3" placeholder="Em muốn chia sẻ điều gì với cả lớp..."></textarea>

<script>
    let posts = JSON.parse(localStorage.getItem('classPosts')) || [];

    function saveAndRender() {
        localStorage.setItem('classPosts', JSON.stringify(posts));
        renderPosts();
    }

    function addPost() {
        const nameInput = document.getElementById('userName');
        const contentInput = document.getElementById('postContent');

        if (!nameInput.value || !contentInput.value) {
            alert("Em vui lòng nhập đầy đủ tên và nội dung nhé!");
            return;
        }

        const newPost = {
            id: Date.now(),
            author: nameInput.value,
            content: contentInput.value,
            time: new Date().toLocaleString('vi-VN'),
            likes: 0,
            comments: []
        };

        posts.unshift(newPost);
        contentInput.value = '';
        saveAndRender();
    }

    function likePost(id) {
        const post = posts.find(p => p.id === id);
        post.likes++;
        saveAndRender();
    }

    function addComment(event, id) {
        if (event.key === 'Enter' && event.target.value) {
            const post = posts.find(p => p.id === id);
            post.comments.push(event.target.value);
            event.target.value = '';
            saveAndRender();
        }
    }

    function renderPosts() {
        const feed = document.getElementById('feed');
        feed.innerHTML = posts.map(post => `
            <div class="post">
                <div class="post-header">
                    <span class="author">👤 ${post.author}</span>
                    <span class="time">${post.time}</span>
                </div>
                <div class="content">${post.content}</div>
                <div class="actions">
                    <button class="btn-like" onclick="likePost(${post.id})">
                        ❤️ ${post.likes} Yêu thích
                    </button>
                </div>
                <div class="comment-section">
                    <div id="comments-${post.id}">
                        ${post.comments.map(c => <div class="comment">💬 ${c}</div>).join('')}
                    </div>
                    <input type="text" placeholder="Viết phản hồi... (nhấn Enter)" 
                           onkeypress="addComment(event, ${post.id})" 
                           style="margin-top:10px; font-size: 12px; padding: 8px;">
                </div>
            </div>
        `).join('');
    }

    // Khởi tạo trang web
    renderPosts();
</script>
